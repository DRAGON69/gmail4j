18  package com.googlecode.gmail4j.rss;
19  
20  import java.net.Authenticator;
21  import java.net.URL;
22  import java.util.ArrayList;
23  import java.util.List;
24  
25  import com.googlecode.gmail4j.GmailClient;
26  import com.googlecode.gmail4j.GmailException;
27  import com.googlecode.gmail4j.GmailMessage;
28  import com.googlecode.gmail4j.auth.Credentials;
29  import com.googlecode.gmail4j.auth.GmailHttpAuthenticator;
30  import com.sun.syndication.feed.synd.SyndEntry;
31  import com.sun.syndication.feed.synd.SyndFeed;
32  import com.sun.syndication.io.SyndFeedInput;
33  import com.sun.syndication.io.XmlReader;
34  
35  /**
36   * Gmail RSS feed based {@link GmailClient} implementation.
37   * <p>
38   * Functionality is limited to unread message retrieval.
39   * <p>
40   * Example use:
41   * <p><blockquote><pre>
42   *     GmailClient client = new RssGmailClient();
43   *     client.setLoginCredentials(new Credentials("user", "pass".toCharArray()));
44   *     client.init();
45   *     for (GmailMessage message : client.getUnreadMessages()) {
46   *         System.out.println(message.getFrom() + ": " + message.getTitle());
47   *     }
48   * </pre></blockquote>
49   * 
50   * @see RssGmailMessage
51   * @see GmailClient
52   * @see Credentials
53   * @author Tomas Varaneckas &lt;tomas.varaneckas@gmail.com&gt;
54   * @version $Id: RssGmailClient.java 11 2009-03-20 12:31:30Z tomas.varaneckas $
55   * @since 0.1
56   */
57  public class RssGmailClient extends GmailClient {
58      
59      /**
60       * Gmail feed URL. It should remain as is, however you can set a new value.
61       * 
62       * @see #setGmailFeedUrl(String)
63       */
64      private String gmailFeedUrl = "https://mail.google.com/mail/feed/atom";
65  
66      @Override
67      public void init() {
68          loginCredentials.validate();
69          Authenticator.setDefault(new GmailHttpAuthenticator(loginCredentials));
70      }
71  
72      /**
73       * Gets {@link #gmailFeedUrl}
74       * 
75       * @return Gmail feed URL
76       */
77      public String getGmailFeedUrl() {
78          return gmailFeedUrl;
79      }
80  
81      /**
82       * Sets {@link #gmailFeedUrl}
83       * <p>
84       * Use it only if you would like to use another Gmail feed URL than the 
85       * default <a href="https://mail.google.com/mail/feed/atom">
86       * https://mail.google.com/mail/feed/atom</a>.
87       * 
88       * @param gmailFeedUrl New Gmail feed URL
89       */
90      public void setGmailFeedUrl(final String gmailFeedUrl) {
91          this.gmailFeedUrl = gmailFeedUrl;
92      }
93      
94      @Override
95      public List<GmailMessage> getUnreadMessages() {
96          final List<GmailMessage> messages = new ArrayList<GmailMessage>();
97          try {
98              final URL feedSource = new URL(gmailFeedUrl);
99              final SyndFeedInput feedInput = new SyndFeedInput();
100             final SyndFeed gmail = feedInput.build(new XmlReader(feedSource));
101             for (final Object entry : gmail.getEntries()) {
102                 if (entry instanceof SyndEntry) {
103                     messages.add(new RssGmailMessage((SyndEntry) entry));
104                 }
105             }
106             if (log.isDebugEnabled()) {
107                 log.debug("Got " + messages.size() + " new messages.");
108             }
109         } catch (final Exception e) {
110             throw new GmailException("Failed getting unread messages", e);
111         }
112         return messages;
113     }
114 
115     @Override
116     protected void finalize() throws Throwable {
117         loginCredentials.dispose();
118         super.finalize();
119     }
120     
121 }
