43  /**
44   * Dialog for getting {@link Credentials}.
45   * <p>
46   * Designed mainly for testing purposes.
47   * <p>
48   * Example use:
49   * <p><blockquote><pre>
50   *     Credentials login = LoginDialog.show("Gmail Login");
51   * </pre></blockquote>

54   * @author Tomas Varaneckas &lt;tomas.varaneckas@gmail.com&gt;
55   * @version $Id: LoginDialog.java 23 2009-03-24 05:49:42Z tomas.varaneckas $
56   * @since 0.1
57   */
58  public class LoginDialog {
59         
60      /**
61       * Dialog frame
62       */
63      private JFrame dialog;
64      
65      /**
66       * User text input
67       */
68      private JTextField user;
69      
70      /**
71       * Password input
72       */
73      private JPasswordField pass;

75      /**
76       * Username label
77       */
78      private JLabel labelUser;

80      /**

83      private JLabel labelPass;

87       */
88      private JButton ok;
89  
90      /**
91       * Countdown latch to make this dialog application modal

93      private CountDownLatch latch;

108                 instance = new LoginDialog();

119     /**
120      * Method that shows login window with prompt for username and password,
121      * like this:
122      * <p><pre>
123      * +------ prompt ------+
124      * |Username            |
125      * |+------------------+|
126      * ||                  ||
127      * |+------------------+|
128      * |Password            |
129      * |+------------------+|
130      * ||                  ||
131      * |+------------------+|
132      * |[OK]                |
133      * +--------------------+
134      * </pre>
135      * @param prompt
136      * @return Credentials with username/password from the dialog prompt
137      */
138     public Credentials show(final String prompt) {
139         createDialog(prompt);
140         final Credentials login = new Credentials();
141         createListeners(login);
142         showDialog();
143         waitForData();
144         return login;
145     }
146 
147     /**
148      * Creates the required listeners
149      * 
150      * @param login Credentials object for filling the data
151      */
152     private void createListeners(final Credentials login) {
153         pass.addKeyListener(new KeyAdapter() {
154             @Override
155             public void keyReleased(final KeyEvent e) {
156                 if (e.getKeyCode() == KeyEvent.VK_ENTER) {
157                     setCredentials(login);
158                     dialog.setVisible(false);
159                 }
160             } 
161         });
162         ok.addActionListener(new ActionListener() {
163             public void actionPerformed(final ActionEvent e) {
164                 setCredentials(login);
165                 dialog.setVisible(false);
166             }
167         });
168         dialog.addWindowListener(new WindowAdapter() {
169             @Override
170             public void windowClosing(final WindowEvent e) {
171                 log.warn("Login dialog closed, no username/password specified");
172                 latch.countDown();
173             }
174         });
175     }
176 
177     /**
178      * Lets latch to wait for data
179      */
180     private void waitForData() {
181         try {
182             log.info("Waiting for username/password");
183             latch.await(30, TimeUnit.SECONDS);
184         } catch (final InterruptedException e1) {
185             log.warn("Interrupted", e1);
186         }
187     }
188 
189     /**
190      * Creates the dialog
191      * 
192      * @param prompt Dialog prompt
193      */
194     private void createDialog(final String prompt) {
195         dialog = new JFrame(prompt);
196         dialog.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
197         latch = new CountDownLatch(1);
198         labelUser = new JLabel("Username");
199         labelPass = new JLabel("Password");
200         user = new JTextField();
201         pass = new JPasswordField();
202         ok = new JButton("OK");
203     }
204 
205     /**
206      * Shows the dialog
207      */
208     private void showDialog() {
209         final BoxLayout layout = new BoxLayout(dialog.getContentPane(), 
210                 BoxLayout.Y_AXIS);
211         dialog.getContentPane().setLayout(layout);
212         dialog.getContentPane().add(labelUser);
213         dialog.getContentPane().add(user);
214         dialog.getContentPane().add(labelPass);
215         dialog.getContentPane().add(pass);
216         dialog.getContentPane().add(ok);
217         dialog.pack();
218         dialog.setSize(200, dialog.getSize().height);
219         dialog.setResizable(false);
220         dialog.setAlwaysOnTop(true);
221         center(dialog);
222         dialog.setVisible(true);
223         user.requestFocusInWindow();
224     }
225 
226     /**
227      * Centers the frame on screen
228      * 
229      * @param frame Target frame
230      */
231     private void center(final JFrame frame) {
232         final Rectangle screen = GraphicsEnvironment.getLocalGraphicsEnvironment()
233                 .getDefaultScreenDevice().getDefaultConfiguration().getBounds();
234         frame.setLocation(screen.width / 2 - dialog.getSize().width / 2, 
235                 screen.height / 2 - dialog.getSize().height / 2);
236     }
237 
238     /**
239      * Inner method that releases the {@link CountDownLatch} and fills 
240      * {@link Credentials} object with values from Username and Password fields
241      * 
242      * @param user User {@link JTextField}
243      * @param pass Password {@link JPasswordField}
244      * @param login Login {@link Credentials}
245      */
246     private void setCredentials(final Credentials login) {
247         login.setUsername(user.getText());
248         login.setPassword(pass.getPassword());
249         latch.countDown();
250     }
