61      /**
62       * Tests retrieval of unread messages
63       */
64      @Test
65      public void testGetUnreadMessages() {
66          try {
67              final ImapGmailClient client = new ImapGmailClient();
68              final ImapGmailConnection connection = new ImapGmailConnection();
69              connection.setLoginCredentials(conf.getGmailCredentials());
70              if (conf.useProxy()) {
71                  connection.setProxy(conf.getProxyHost(), conf.getProxyPort());
72                  connection.setProxyCredentials(conf.getProxyCredentials());
73              }
74              log.debug("Getting unread messages");
75              client.setConnection(connection);
76              final List<GmailMessage> messages = client.getUnreadMessages();
77              for (GmailMessage message : messages) {
78                  log.debug(message);
79              }
80              assertNotNull("Messages are not null", messages);
81              connection.disconnect();
82          } catch (final Exception e) {
83              log.error("Test Failed", e);
84              fail("Caught exception: " + e.getMessage());
85          }
86      }
87      
88      /**
89       * Test a sending of a simple message
90       */
91      @Test
92      public void testSendMessage() {
93          final GmailClient client = new ImapGmailClient();
94          final GmailConnection connection = new ImapGmailConnection();
95          connection.setLoginCredentials(conf.getGmailCredentials());
96          if (conf.useProxy() && (connection instanceof ProxyAware)) {
97              ((ProxyAware) connection).setProxy(
98                      conf.getProxyHost(), conf.getProxyPort());
99              ((ProxyAware) connection).setProxyCredentials(
100                     conf.getProxyCredentials());
101         }
102         client.setConnection(connection);
103         GmailMessage msg = new JavaMailGmailMessage();
104         msg.setSubject("Test mail subject. Unicode: ąžuolėlį");
105         msg.setContentText("Test mail content. Unicode: ąžuolėlį");
106         msg.addTo(new EmailAddress(conf.getTestRecipient()));
107         client.send(msg);
108     }
