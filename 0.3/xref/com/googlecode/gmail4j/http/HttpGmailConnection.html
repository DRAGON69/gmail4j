17  package com.googlecode.gmail4j.http;
18  
19  import java.io.IOException;
20  import java.net.Authenticator;
21  import java.net.InetSocketAddress;
22  import java.net.MalformedURLException;
23  import java.net.Proxy;
24  import java.net.URL;
25  import java.net.URLConnection;
26  import java.net.Proxy.Type;
27  
28  import com.googlecode.gmail4j.GmailConnection;
29  import com.googlecode.gmail4j.GmailException;
30  import com.googlecode.gmail4j.auth.Credentials;
31  import com.googlecode.gmail4j.auth.GmailHttpAuthenticator;
32  
33  /**
34   * {@link GmailConnection} implementation that uses HTTP transport. 
35   * <p>
36   * It is {@link Proxy} aware. Example use:
37   * <p><blockquote><pre>
38   *     GmailClient client = //...
39   *     GmailConnection conn = new HttpGmailConnection(gmailUser, gmailPass);
40   *     client.setConnection(conn);
41   * </pre></blockquote><p>
42   * When using with {@link Proxy}:
43   * <p><blockquote><pre>
44   *     GmailClient client = //...
45   *     HttpGmailConnection conn = new HttpGmailConnection(gmailUser, gmailPass);
46   *     conn.setProxy(proxyHost, proxyPort);
47   *     conn.setProxyCredentials(proxyUser, proxyPass);
48   *     client.setConnection(conn);
49   * </pre></blockquote>
50   * 
51   * @see Credentials
52   * @see GmailConnection
53   * @see Proxy
54   * @see ProxyAware
55   * @author Tomas Varaneckas &lt;tomas.varaneckas@gmail.com&gt;
56   * @version $Id: HttpGmailConnection.java 37 2009-07-09 17:26:41Z tomas.varaneckas $
57   * @since 0.2
58   */
59  public class HttpGmailConnection extends GmailConnection implements ProxyAware {
60      
61      /**
62       * Argless constructor.
63       * <p>
64       * Set credentials manually with {@link #setLoginCredentials(Credentials)}
65       * or {@link #setLoginCredentials(String, char[])}.
66       */
67      public HttpGmailConnection() {
68          //nothing to do
69      }
70      
71      /**
72       * Constructor with Gmail {@link Credentials}
73       * 
74       * @param loginCredentials Gmail login
75       * @throws GmailException if provided credentials are empty
76       */
77      public HttpGmailConnection(final Credentials loginCredentials) {
78          super(loginCredentials);
79      }
80      
81      /**
82       * Convenience constructor with Gmail {@link Credentials}
83       * 
84       * @param username Gmail username
85       * @param password Gmail password
86       * @throws GmailException if provided credentials are empty
87       */    
88      public HttpGmailConnection(final String username, final char[] password) {
89          super(username, password);
90      }
91  
92      /**
93       * HTTP connection URL
94       * 
95       * @see #setUrl(String)
96       */
97      private URL url;
98      
99      /**
100      * HTTP Proxy for making a connection.
101      * 
102      * @see #setProxy(Proxy)
103      * @see #setProxy(String, int)
104      * @see #getProxy()
105      */
106     private Proxy proxy = null;    
107     
108     /**
109      * HTTP Proxy {@link Credentials}
110      * 
111      * @see #proxy
112      * @see #setProxyCredentials(Credentials)
113      * @see #setProxyCredentials(String, char[])
114      */
115     private Credentials proxyCredentials = null;
116     
117     /**
118      * If this is set to true, HttpGmailConnection will skip a call to
119      * <code>Authenticator.setDefault(new GmailHttpAuthenticator);</code>
120      * <p>
121      * Default is <code>false</code>
122      * 
123      * @see #useCustomAuthenticator(boolean)
124      */
125     private boolean useCustomAuthenticator = false;
126     
127     /**
128      * Sets {@link #useCustomAuthenticator(boolean)} flag. By default it's 
129      * set to <code>false</code>
130      * 
131      * @param use skip setting of default {@link Authenticator}?
132      */
133     public void useCustomAuthenticator(final boolean use) {
134         useCustomAuthenticator = use;
135     }
136     
137     /**
138      * Sets {@link #url} for this HTTP connection
139      * 
140      * @param url URL of Gmail service you want to connect to
141      * @throws GmailException if URL is malformed
142      */
143     public void setUrl(final String url) {
144         try {
145             this.url = new URL(url);
146         } catch (final MalformedURLException e) {
147             throw new GmailException("Failed creating Gmail connection", e);
148         }
149     }
150     
151     /**
152      * Opens {@link URLConnection} for getting data from Gmail RSS.
153      * <p>
154      * May use {@link Proxy} if one is defined
155      * 
156      * @see #proxy
157      * @see #useCustomAuthenticator
158      * @return connection
159      * @throws IOException if opening a connection fails
160      */
161     public URLConnection openConnection() {
162         try {
163             if (!useCustomAuthenticator) {
164                 Authenticator.setDefault(
165                         new GmailHttpAuthenticator(loginCredentials, proxyCredentials));
166             }
167             if (proxy != null) {
168                 return url.openConnection(proxy); 
169             } else {
170                 return url.openConnection();
171             }
172         } catch (final Exception e) {
173             throw new GmailException("Failed opening Gmail connection", e);
174         }
175     }
176     
177     /**
178      * Sets the {@link #proxy}
179      * 
180      * @param proxy Proxy for RSS connection
181      */
182     public void setProxy(final Proxy proxy) {
183         this.proxy = proxy;
184     }
185     
186     /**
187      * A convenience method for setting HTTP {@link #proxy}
188      * 
189      * @param proxyHost Proxy host
190      * @param proxyPort Proxy port
191      */
192     public void setProxy(final String proxyHost, final int proxyPort) {
193         this.proxy = new Proxy(Type.HTTP, InetSocketAddress
194                 .createUnresolved(proxyHost, proxyPort));
195     }
196     
197     /**
198      * Sets {@link #proxyCredentials}
199      * 
200      * @param proxyCredentials Proxy authentication
201      */
202     public void setProxyCredentials(final Credentials proxyCredentials) {
203         this.proxyCredentials = proxyCredentials;
204     }
205     
206     /**
207      * A convenience method for setting {@link #proxyCredentials}
208      * 
209      * @param username Proxy auth username
210      * @param password Proxy auth password
211      */
212     public void setProxyCredentials(final String username, final char[] password) {
213         setProxyCredentials(new Credentials(username, password));
214     }
215     
216     /**
217      * Gets the {@link #proxy}
218      * 
219      * @return Proxy or null if unavailable
220      */
221     public Proxy getProxy() {
222         return this.proxy;
223     }
224     
225     @Override
226     protected void finalize() throws Throwable {
227         loginCredentials.dispose();
228         if (proxyCredentials != null) {
229             proxyCredentials.dispose();
230         }
231         super.finalize();
232     }
233 }
