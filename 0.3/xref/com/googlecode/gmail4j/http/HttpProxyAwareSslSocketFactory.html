69      @Override
70      public Socket createSocket() throws IOException {
71          //FIXME won't work..
72          log.debug("Creating socket! with proxy: " + proxy.address());
73          InetSocketAddress addr = (InetSocketAddress) proxy.address();
74          ProxyClient proxyClient = new ProxyClient();
75          proxyClient.getHostConfiguration().setHost("imap.gmail.com", 993);
76          proxyClient.getHostConfiguration().setProxy(addr.getHostName(), 
77                  addr.getPort());
78          if (proxyCredentials != null) {
79              proxyClient.getState().setProxyCredentials(
80                      AuthScope.ANY, 
81                      new UsernamePasswordCredentials(proxyCredentials.getUsername(), 
82                          new String(proxyCredentials.getPasword())));
83          }
84          log.debug("Trying to connect to proxy");
85          ProxyClient.ConnectResponse resp = proxyClient.connect();
86          if (resp.getConnectMethod().getStatusCode() != HttpStatus.SC_OK) {
87              log.error("Failed to connect. " + resp.getConnectMethod().getStatusLine());
88              throw new GmailException("Failed connecting to IMAP through proxy: " 
89                      + resp.getConnectMethod().getStatusLine());
90          }
91          log.debug("Connected, returning socket");
92          return resp.getSocket();
93      }
