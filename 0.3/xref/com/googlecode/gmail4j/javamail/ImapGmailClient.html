33  /**
34   * JavaMail IMAP based {@link GmailClient}
35   * <p>
36   * Example of getting unread messages:
37   * <p><blockquote><pre>
38   *     GmailConnection conn = new ImapGmailConnection();
39   *     //configure connection
40   *     GmailClient client = new ImapGmailClient();
41   *     client.setConnection(conn);
42   *     List&lt;GmailMessage&gt; unreadMessages = client.getUnreadMessages();
43   * </pre></blockquote><p>
44   * Example of sending a simple message:
45   * <p><blockquote><pre>
46   *     GmailConnection conn = new ImapGmailConnection();
47   *     //configure connection
48   *     GmailClient client = new ImapGmailClient();
49   *     client.setConnection(conn);
50   *     GmailMessage message = new JavaMailGmailMessage();
51   *     message.setSubject("Hi!");
52   *     message.setContentText("A message from Gmail4J");
53   *     message.addTo(new EmailAddress("j.smith@example.com"));  
54   *     client.send(message);   
55   * </pre></blockquote></p>
56   * 
57   * @see GmailClient
58   * @see ImapGmailConnection
59   * @author Tomas Varaneckas &lt;tomas.varaneckas@gmail.com&gt;
60   * @version $Id: ImapGmailClient.java 37 2009-07-09 17:26:41Z tomas.varaneckas $
61   * @since 0.3
62   */
63  public class ImapGmailClient extends GmailClient {
64  
65      @Override
66      public List<GmailMessage> getUnreadMessages() {
67          try {
68              final List<GmailMessage> unread = new ArrayList<GmailMessage>();
69              final Store store = openGmailStore();
70              final Folder folder = store.getFolder("INBOX");
71              folder.open(Folder.READ_ONLY); 
72              for (final Message msg : folder.search(new FlagTerm(
73                      new Flags(Flags.Flag.SEEN), false))) {
74                  unread.add(new JavaMailGmailMessage(msg));
75              }
76              return unread;
77          } catch (final Exception e) {
78              throw new GmailException("Failed getting unread messages", e);
79          }
80      }
81      
82      /**
83       * Opens Gmail {@link Store}
84       * 
85       * @return instance of Gmail store
86       * @throws GmailException if GmailConnection is not ImapGmailConnection
87       */
88      private Store openGmailStore() {
89          if (connection instanceof ImapGmailConnection) {
90              return ((ImapGmailConnection) connection).openGmailStore();
91          } 
92          throw new GmailException("ImapGmailClient requires ImapGmailConnection!");
93      }
94      
95      /**
96       * Gets Gmail {@link Transport} for sending messages
97       * 
98       * @return Configured Gmail Transport ready for use
99       * @throws GmailException if GmailConnection is not ImapGmailConnection
100      */    
101     private Transport getGmailTransport() {
102         if (connection instanceof ImapGmailConnection) {
103             return ((ImapGmailConnection) connection).getTransport();
104         }
105         throw new GmailException("ImapGmailClient requires ImapGmailConnection!");
106     }
107 
108     @Override
109     public void send(final GmailMessage message) {
110         if (message instanceof JavaMailGmailMessage) {
111             try {
112                 final JavaMailGmailMessage msg = (JavaMailGmailMessage) message;
113                 getGmailTransport().sendMessage(
114                         msg.getMessage(), 
115                         msg.getMessage().getAllRecipients());
116             } catch (final Exception e) {
117                 throw new GmailException("Failed sending message: " + message, e);
118             }
119         } else {
120             throw new GmailException("ImapGmailClient requires JavaMailGmailMessage!");
121         }
122     }
