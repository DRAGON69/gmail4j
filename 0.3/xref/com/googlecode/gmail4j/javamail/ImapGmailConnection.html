39  /**
40   * JavaMail based IMAP {@link GmailConnection} implementation.
41   * <p>
42   * ImapGmailConnection is {@link ProxyAware}, allowing to use JavaMail IMAP
43   * via HTTP proxy, which was not possible by JavaMail design.
44   * <p>
45   * Example:
46   * <p><blockquote><pre>
47   *     GmailConnection conn = new ImapGmailConnection();
48   *     conn.setLoginCredentials(new Credentials("user", "pass".toCharArray()));
49   *     //if proxy is required
50   *     ((ProxyAware) conn).setProxy("proxy.example.com", 8080);
51   *     //if proxy auth is required
52   *     ((ProxyAware) conn).setProxyCredentials("proxyuser", 
53   *         "proxypass".toCharArray());
54   *     GmailClient client = new ImapGmailClient();
55   *     client.setConnection(conn);
56   * </pre></blockquote></p>
57   * 
58   * @author Tomas Varaneckas &lt;tomas.varaneckas@gmail.com&gt;
59   * @version $Id: ImapGmailConnection.java 37 2009-07-09 17:26:41Z tomas.varaneckas $
60   * @since 0.3
61   */

215     /**
216      * Opens Gmail {@link Store}
217      * 
218      * @return singleton instance of Gmail {@link Store}
219      */
220     public Store openGmailStore() {
221         try {
222             store = getSession().getStore("imaps");
223             store.connect(gmailImapHost, loginCredentials.getUsername()
224                     .concat("@gmail.com"), 
225                     new String(loginCredentials.getPasword()));
226         } catch (final Exception e) {
227             throw new GmailException("Failed opening Gmail IMAP store", e);
228         }
229         return store;
230     }

237     public Transport getTransport() {
238         try {
239             final Transport transport = getSession().getTransport();
240             transport.connect(loginCredentials.getUsername(), 
241                     new String(loginCredentials.getPasword()));
242             return transport;
243         } catch (final Exception e) {
244             throw new GmailException("ImapGmailClient requires ImapGmailConnection!");

248     /**
249      * Gets Gmail {@link Session}
250      * 
251      * @return Configured and ready for use Gmail {@link Session}
252      */
253     public Session getSession() {
254         synchronized (this) {
255             if (mailSession == null) {
256                 properties = System.getProperties();
257                 properties.put("mail.transport.protocol", "smtps");
258                 properties.put("mail.debug", true);
259                 properties.put("mail.smtps.host", gmailSmtpHost);
260                 properties.put("mail.smtps.port", gmailSmtpPort);
261                 if (proxy != null) {
262                     log.debug("Setting proxy: " + proxy.address());
263                     final SocketFactory sf = 
264                         new HttpProxyAwareSslSocketFactory(proxy, proxyCredentials);
265                     properties.put("mail.imap.host", gmailImapHost);
266                     properties.put("mail.imaps.ssl.socketFactory", sf);
267                     properties.put("mail.imaps.ssl.socketFactory.fallback", "false");
268                     properties.put("mail.imaps.ssl.socketFactory.port", "993");
269                 }
270                 if (proxyCredentials != null) {
271                     mailSession = Session.getInstance(properties);               
272                 } else {
273                     mailSession = Session.getInstance(properties);
274                 }
275                 for (Provider p : mailSession.getProviders()) {
276                     log.debug(p);
277                 }
278             }
279         }
280         return mailSession;
281     }

283     /**
284      * Disconnects from Gmail
285      */
286     public void disconnect() {
287         try {
288             if (store != null) {
289                 store.close();
290             }
291         } catch (final Exception e) {
292             log.warn("Failed closing Gmail IMAP store", e);
293         }
294     }

308     public void setProxyCredentials(final Credentials credentials) {
