35  /**
36   * <a href="http://java.sun.com/products/javamail/">JavaMail</a> implementation 
37   * of {@link GmailMessage}
38   * <p>
39   * Example: Send a simple message:
40   * <p><blockquote><pre>
41   *     GmailClient client = new ImapGmailClient();
42   *     //configure the client...
43   *     GmailMessage message = new JavaMailGmailMessage();
44   *     message.setSubject("Subject");
45   *     message.setContentText("Content");
46   *     message.addTo(new EmailAddress("j.smith@example.com");
47   *     client.send(message);
48   * </pre></blockquote></p> 
49   * 
50   * @author Tomas Varaneckas &lt;tomas.varaneckas@gmail.com&gt;
51   * @version $Id: JavaMailGmailMessage.java 37 2009-07-09 17:26:41Z tomas.varaneckas $
52   * @since 0.3
53   */
54  public class JavaMailGmailMessage extends GmailMessage {
55  
56      /**
57       * Original JavaMail {@link Message}
58       */
59      final Message source;
60      
61      /**
62       * Cache for {@link #toString()}
63       */
64      private StringBuilder toString;
65      
66      /**
67       * Sender's email address
68       */
69      private EmailAddress from;
70      
71      /**
72       * Constructor with source {@link Message}
73       * 
74       * @param source JavaMail message with data
75       */
76      public JavaMailGmailMessage(final Message source) {
77          this.source = source;
78      }
79      
80      /**
81       * Constructor that creates a new empty JavaMail {@link MimeMessage}
82       */
83      public JavaMailGmailMessage() {
84          this.source = new MimeMessage((Session) null);
85      }
86  
87      /**
88       * Gets the {@link #source} {@link Message}
89       * 
90       * @return source message
91       */
92      public Message getMessage() {
93          return source;
94      }
95      
96      @Override
97      public void addTo(final EmailAddress to) {
98          try {
99              if (to.hasName()) {
100                 source.addRecipient(RecipientType.TO, 
101                         new InternetAddress(to.getEmail(), to.getName()));
102             } else {
103                 source.addRecipient(RecipientType.TO, 
104                         new InternetAddress(to.getEmail()));
105             }
106         } catch (final Exception e) {
107             throw new GmailException("Failed adding To recipient", e);
108         }
109     }
110     
111     @Override
112     public List<EmailAddress> getTo() {
113         try {
114             return getAddresses(RecipientType.TO);
115         } catch (final Exception e) {
116             throw new GmailException("Failed getting List of To recipients", e);
117         }
118     }
119     
120     @Override
121     public List<EmailAddress> getCc() {
122         try {
123             return getAddresses(RecipientType.CC);
124         } catch (final Exception e) {
125             throw new GmailException("Failed getting List of Cc recipients", e);
126         }
127     }    
128     
129     /**
130      * Gets a {@link List} of {@link EmailAddress} by {@link RecipientType}
131      * 
132      * @param type Recipient type
133      * @return List of Addresses
134      * @throws MessagingException in case something is wrong
135      */
136     private List<EmailAddress> getAddresses(final RecipientType type) 
137             throws MessagingException {
138         final List<EmailAddress> addresses = new ArrayList<EmailAddress>();
139         for (final Address addr : source.getRecipients(type)) {
140             final InternetAddress temp = (InternetAddress) addr;
141             addresses.add(new EmailAddress(temp.getPersonal(), temp.getAddress()));
142         }
143         return addresses;
144     }
145 
146     @Override
147     public void setFrom(final EmailAddress from) {
148         try {
149             if (from.hasName()) {
150                 source.setFrom(new InternetAddress(
151                         from.getEmail(), from.getName()));
152             } else {
153                 source.setFrom(new InternetAddress(from.getEmail()));
154             }
155         } catch (final Exception e) {
156             throw new GmailException("Failed setting from address", e);
157         }
158     }
159     
160     @Override
161     public EmailAddress getFrom() {
162         if (from == null) {
163             try {
164                 final InternetAddress f = (InternetAddress) source.getFrom()[0];
165                 from = new EmailAddress(f.getPersonal(), f.getAddress());
166             } catch (final Exception e) {
167                 throw new GmailException("Failed getting from address", e);
168             }
169         }
170         return from;
171     }
172     
173     @Override
174     public Date getSendDate() {
175         try {
176             return source.getSentDate();
177         } catch (final Exception e) {
178             throw new GmailException("Failed getting send date", e);
179         }
180     }
181     
182     @Override
183     public void setSubject(final String subject) {
184         try {
185             source.setSubject(subject);
186         } catch (final Exception e) {
187             throw new GmailException("Failed setting subject", e);
188         }
189     }
190 
191     @Override
192     public String getSubject() {
193         try {
194             return source.getSubject();
195         } catch (final Exception e) {
196             throw new GmailException("Failed getting message subject", e);
197         }
198     }
199     
200     @Override
201     public void setContentText(final String contentText) {
202         try {
203             source.setText(contentText);
204         } catch (final Exception e) {
205             throw new GmailException("Failed settting content text", e);
206         }
207     }
208     
209     @Override
210     public String getPreview() {
211         try {
212             return source.getContent().toString();
213         } catch (final Exception e) {
214             throw new GmailException("Failed getting message preview", e);
215         }
216     }
217     
218     @Override
219     public String toString() {
220         if (toString != null) {
221             return toString.toString();
222         }
223         try {
224             toString = new StringBuilder();
225             toString.append("MailMessage:{from:").append(getFrom())
226                 .append(";sendDate:").append(getSendDate())
227                 .append(";subject:").append(getSubject())
228                 .append(";preview:").append(getPreview()).append(";}");
229             return toString.toString();
230         } catch (final Exception e) {
231             toString = null;
232             return super.toString().concat("(e:").concat(e.getMessage())
233                     .concat(")");
234         }
235     }
