17  package com.googlecode.gmail4j.auth;
18  
19  import com.googlecode.gmail4j.GmailException;
20  import com.googlecode.gmail4j.util.LoginDialog;
21  
22  /**
23   * Login credentials containing username and password.
24   * <p>
25   * This class stores password as a <code>char[]</code> and allows you to dispose
26   * the array via {@link #dispose()} to avoid sniffing.
27   * <p>
28   * There are several ways to construct your <code>Credentials</code>:
29   * <p><blockquote><pre>
30   *     String username = ...
31   *     char[] password = ...
32   *     Credentials login = new Credentials(username, password);
33   *     //password char array is cleared automatically
34   * </pre></blockquote><p>
35   * Another approach:
36   * <p><blockquote><pre>
37   *     String username = ...
38   *     char[] password = ...
39   *     Credentials login = new Credentials();
40   *     login.setUsername(username);
41   *     login.setPassword(password)
42   *     //password char array is cleared automatically
43   * </pre></blockquote><p>
44   * For testing purposes you can construct <code>Credentials</code> with help of 
45   * {@link LoginDialog#show(String)}
46   * <p><blockquote><pre>
47   *     Credentials login = LoginDialog.show("Gmail Login");
48   * </pre></blockquote><p>
49   * There are more methods of interest: 
50   * <ul><li>{@link #validate()} checks if
51   * username and password were provided and throws {@link GmailException} in case
52   * something is missing.</li>
53   * <li>{@link #dispose()} clears the username and password to prevent further 
54   * use.</li></ul>
55   * 
56   * @author Tomas Varaneckas &lt;tomas.varaneckas@gmail.com&gt;
57   * @version $Id: Credentials.java 37 2009-07-09 17:26:41Z tomas.varaneckas $
58   * @since 0.1
59   */
60  public final class Credentials {
61  
62      /**
63       * Password stored as <code>char[]</code> to enable secure disposal 
64       */
65      private char[] password;
66  
67      /**
68       * Username
69       */
70      private String username;
71  
72      /**
73       * Argless constructor to enable setter based configuration.
74       * <p>
75       * You will have to manually call {@link #setUsername(String)} and 
76       * {@link #setPassword(char[])} (nd optionally {@link #validate()}
77       * afterwards;
78       */
79      public Credentials() {
80          //nothing to do
81      }
82  
83      /**
84       * Constructor that sets {@link #username} and {@link #password}. Given
85       * <code>password char[]</code> argument gets cleared for security purposes.
86       * <p>
87       * It also calls {@link #validate()} automatically to check if user/pass
88       * is not empty. 
89       * 
90       * @param username Username
91       * @param password Password
92       */
93      public Credentials(final String username, final char[] password) {
94          setUsername(username);
95          setPassword(password);
96          validate();
97      }
98      
99      /**
100      * Validates existence of {@link #username} and {@link #password} properties.
101      * 
102      * @throws GmailException in case username and/or password are missing.
103      */
104     public void validate() {
105         if (username == null || username.length() == 0 
106                 || password == null || password.length == 0) {
107             throw new GmailException("No username and/or password provided");
108         }
109     }
110  
111     /**
112      * Sets {@link #username}
113      * 
114      * @param username Username
115      */
116     public void setUsername(final String username) {
117         this.username = username;
118     }
119 
120     /**
121      * Sets {@link #password} and clears the given <code>char[]</code> argument
122      * for security purposes.
123      * 
124      * @param password Password
125      */
126     public void setPassword(final char[] password) {
127         this.password = new char[password.length];
128         System.arraycopy(password, 0, this.password, 0, password.length);
129         disposePassword(password);
130     }
131 
132     /**
133      * Disposes a given <code>char[]</code>. A way to destroy password arguments.
134      * 
135      * @param password Password for disposal
136      */
137     private void disposePassword(final char[] password) {
138         for (int i = 0; i < password.length; i++) {
139             password[i] = 0;
140         }
141     }
142 
143     /**
144      * Gets {@link #password}
145      * @return A reference to password <code>char[]</code>
146      */
147     public char[] getPasword() {
148         return this.password;
149     }
150 
151     /**
152      * Gets {@link #username}
153      * 
154      * @return Username
155      */
156     public String getUsername() {
157         return this.username;
158     }
159     
160     /**
161      * Disposes current <code>Credentials</code> instance by clearing 
162      * {@link #username} and {@link #password}.
163      * <p>
164      * Object can be reused again with new {@link #username} and 
165      * {@link #password}
166      * 
167      * @see #disposePassword(char[])
168      */
169     public void dispose() {
170         this.username = null;
171         disposePassword(this.password);
172         this.password = null;
173     }
174     
175     @Override
176     protected void finalize() throws Throwable {
177         dispose();
178         super.finalize();
179     }
180 
181 }
