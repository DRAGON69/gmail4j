172     @Override
173     public List<GmailMessage> getUnreadMessages() {
174         try {
175             final List<GmailMessage> unread = new ArrayList<GmailMessage>();
176             final Store store = openGmailStore();
177             final Folder folder = getFolder(this.srcFolder,store);
178             folder.open(Folder.READ_ONLY);
179             for (final Message msg : folder.search(new FlagTerm(
180                     new Flags(Flags.Flag.SEEN), false))) {
181                 unread.add(new JavaMailGmailMessage(msg));
182             }
183             return unread;
184         } catch (final Exception e) {
185             throw new GmailException("Failed getting unread messages", e);
186         }
187     }

215     @Override
216     public void send(final GmailMessage message) {
217         if (message instanceof JavaMailGmailMessage) {
218             Transport transport = null;
219             
220             try {
221                 final JavaMailGmailMessage msg = (JavaMailGmailMessage) message;
222                 transport = getGmailTransport();
223                 transport.sendMessage(
224                         msg.getMessage(),
225                         msg.getMessage().getAllRecipients());
226             } catch (final Exception e) {
227                 throw new GmailException("Failed sending message: " + message, e);
228             }
229             finally{
230                 if(transport.isConnected())
231                 {
232                     try {
233                         transport.close();
234                     } catch (final Exception e) {
235                         LOG.warn("Cannot Close ImapGmailConnection : " + transport, e);
236                     }
237                 }
238             }
239         } else {
240             throw new GmailException("ImapGmailClient requires JavaMailGmailMessage!");
241         }
242     }

244     /**
245      * Moves given {@link GmailMessage}'s to {@link ImapGmailLabel.Trash} folder.
246      *
247      * @param gmailMessages {@link GmailMessage} message(s)
248      * @throws GmailException if unable to move {@link GmailMessage}'s to
249      * the Trash Folder
250      */
251     public void moveToTrash(final GmailMessage[] gmailMessages) {
252         if (gmailMessages == null || gmailMessages.length <= 0) {
253             LOG.warn("ImapGmailClient requires GmailMessage(s) to move"
254                     + " to move messages to trash folder");
255             return;
256         }
257         Folder folder = null;
258         
259         try {
260             final Store store = openGmailStore();
261             folder = getFolder(this.srcFolder,store);
262             if(!folder.isOpen())
263             {
264                 folder.open(Folder.READ_WRITE);                
265             }
266             
267             List<Message> markedMsgList = new ArrayList<Message>();
268             for (GmailMessage gmailMessage : gmailMessages) {
269                 // get only messages that match to the specified message number
270                 Message message = folder.getMessage(gmailMessage.getMessageNumber());
271                 message.setFlag(Flags.Flag.SEEN, true);
272                 // mark message as delete
273                 message.setFlag(Flags.Flag.DELETED, true);
274                 markedMsgList.add(message);
275             }
276 
277             Folder trash = getFolder(ImapGmailLabel.TRASH.getName(),store);
278             if(folder.getURLName().equals(trash.getURLName())){
279                 LOG.warn("ImapGmailClient trying to move GmailMessage(s) within"
280                         + " same folder(ImapGmailLabel.TRASH.getName())");
281             }
282             // move the marked messages to trash folder
283             if (!markedMsgList.isEmpty()) {
284                 folder.copyMessages(markedMsgList.toArray(new Message[0]), trash);
285             }
286         } catch (Exception e) {
287             throw new GmailException("ImapGmailClient failed moving GmailMessage(s)"
288                     + " to trash folder: " + e);
289         } finally {
290             closeFolder(folder);
291         }
292     }

434     /**
435      * Move {@link GmailMessage} to a given destination folder.
436      *
437      * @param destFolder the destination {@link Folder} name.See {@see ImapGmailLabel}
438      * @param messageNumber the message number ex:{@code gmailMessage.getMessageNumber()}
439      * @throws GmailException if it fails to move {@link GmailMessage} to the
440      * destination folder
441      */
442     public void moveTo(ImapGmailLabel destFolder,int messageNumber){
443         if (messageNumber <= 0) {
444             throw new GmailException("ImapGmailClient invalid GmailMessage number");
445         }
446 
447         Folder fromFolder = null;
448         Folder toFolder = null;
449 
450         try {
451             final Store store = openGmailStore();
452             fromFolder = getFolder(this.srcFolder, store);
453             fromFolder.open(Folder.READ_WRITE);
454             Message message = fromFolder.getMessage(messageNumber);
455 
456             if (message != null) {
457                 toFolder = getFolder(destFolder.getName(), store);
458 
459                 if (fromFolder.getURLName().equals(toFolder.getURLName())) {
460                     throw new GmailException("ImapGmailClient cannot move "
461                             + "GmailMessage within same folder "
462                             + "(from " + fromFolder.getFullName() + " to "
463                             + toFolder.getFullName() + ")");
464                 }
465                 // copy from source folder to destination folder
466                 fromFolder.copyMessages(new Message[]{message}, toFolder);
467                 // move the copied message to trash folder
468                 moveToTrash(new GmailMessage[]{new JavaMailGmailMessage(message)});
469             }
470         } catch (GmailException ge) {
471             throw ge;
472         } catch (Exception e) {
473             throw new GmailException("ImapGmailClient failed moving"
474                     + " GmailMessage from " + fromFolder.getFullName(), e);
475         } finally {
476             closeFolder(fromFolder);
477         }
478     }

491         try {
492             final List<GmailMessage> priorityMessages = new ArrayList<GmailMessage>();
493             final Store store = openGmailStore();
494             Folder folder = getFolder(ImapGmailLabel.IMPORTANT.getName(),store);
495             folder.open(Folder.READ_ONLY); 
496             for (final Message msg : folder.search(new FlagTerm(
497                     new Flags(Flags.Flag.SEEN), !unreadOnly))) {
498                 priorityMessages.add(new JavaMailGmailMessage(msg));
499             }

504         }

507      /**
508      * Return the {@link Folder} object corresponding to the given 
509      * {@link ImapGmailLabel} name. 
510      * Note that a {@link Folder} object is returned only if the named 
511      * {@link Folder} physically exist on the Store.
512      * 
513      * @param name the name of the folder
514      * @param store instance of Gmail {@link Store}
515      */
516     private Folder getFolder(String name, final Store store) {
517         if (store == null) {
518             throw new GmailException("Gmail IMAP store cannot be null");
519         }
520         try {
521             name = ((name == null) ? this.srcFolder : name);
522             Folder folder = store.getFolder(name);
523             if (folder.exists()) {
524                 return folder;
525             }
526         } catch (final Exception e) {
527             throw new GmailException("ImapGmailClient failed getting "
528                     + "Folder: " + name, e);
529         }
530 
531         throw new GmailException("ImapGmailClient Folder name cannot be null");
532     }

541             try {
542                 if (folder.isOpen()) {

544                 } else {
545                     LOG.info(folder.getName() + " folder is already open.");
