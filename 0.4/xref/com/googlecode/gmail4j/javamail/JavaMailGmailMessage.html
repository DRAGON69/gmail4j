100     @Override
101     public void addTo(final EmailAddress to) {
102         try {
103             if (to.hasName()) {
104                 source.addRecipient(RecipientType.TO,
105                         new InternetAddress(to.getEmail(), to.getName()));
106             } else {
107                 source.addRecipient(RecipientType.TO,
108                         new InternetAddress(to.getEmail()));
109             }
110         } catch (final Exception e) {
111             throw new GmailException("Failed adding To recipient", e);
112         }
113     }

133     /**
134      * Gets a {@link List} of {@link EmailAddress} by {@link RecipientType}
135      * 
136      * @param type Recipient type
137      * @return List of Addresses
138      * @throws MessagingException in case something is wrong
139      */
140     private List<EmailAddress> getAddresses(final RecipientType type)
141             throws MessagingException {
142         final List<EmailAddress> addresses = new ArrayList<EmailAddress>();
143         for (final Address addr : source.getRecipients(type)) {
144             final InternetAddress temp = (InternetAddress) addr;
145             addresses.add(new EmailAddress(temp.getPersonal(), temp.getAddress()));
146         }
147         return addresses;
148     }

222     @Override
223     public String toString() {
224         if (toString != null) {
225             return toString.toString();
226         }
227         try {
228             toString = new StringBuilder();
229             toString.append("MailMessage:{from:").append(getFrom())
230                     .append(";sendDate:").append(getSendDate())
231                     .append(";subject:").append(getSubject())
232                     .append(";preview:").append(getPreview()).append(";}");
233             return toString.toString();
234         } catch (final Exception e) {
235             toString = null;
236             return super.toString().concat("(e:").concat(e.getMessage())
237                     .concat(")");
238         }
239     }

250     @Override
251     public MessageHeaderInfo getMessageHeaderInfo() {
252         MessageHeaderInfo headerInfo = null;
253         try {
254             Map<String, String> registry = new HashMap<String, String>();
255 
256             // message header tags used to get header information
257             String[] headers = new String[]{
258                 CommonConstants.MESSAGE_ID,
259                 CommonConstants.MESSAGE_SUBJECT,
260                 CommonConstants.MESSAGE_IN_REPLY_TO,
261                 CommonConstants.MESSAGE_REFERENCES};
262 
263             @SuppressWarnings("unchecked")
264             Enumeration<Header> matchingHeaders =
265                     source.getMatchingHeaders(headers);
266 
267             while (matchingHeaders.hasMoreElements()) {
268                 Header header = matchingHeaders.nextElement();
269                 registry.put(header.getName(), header.getValue());
270             }
271 
272             if (!registry.isEmpty()) {
273                 String messageId = registry.get(CommonConstants.MESSAGE_ID);
274                 String subject = registry.get(CommonConstants.MESSAGE_SUBJECT);
275                 String inReplyTo = registry.get(CommonConstants.MESSAGE_IN_REPLY_TO);
276                 String references = registry.get(CommonConstants.MESSAGE_REFERENCES);
277 
278                 if (messageId != null) {
279                     headerInfo = new MessageHeaderInfo(messageId);
280                     headerInfo.setSubject(subject);
281                     headerInfo.setInReplyTo(inReplyTo);
282 
283                     if (references != null) {
284                         StringTokenizer st = new StringTokenizer(references, "\n");
285                         while (st.hasMoreTokens()) {
286                             String reference = st.nextToken().trim();
287                             headerInfo.addReferences(reference);
288                         }
289                     }
290                 }
291             }
292         } catch (final Exception e) {
293             throw new GmailException("Failed getting message header information", e);
294         }
295         return headerInfo;
296     }
