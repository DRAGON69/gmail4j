42  /**
43   * Dialog for getting {@link Credentials}.
44   * <p>
45   * Designed mainly for testing purposes.
46   * <p>
47   * Example use:
48   * <p><blockquote><pre>
49   *     Credentials login = LoginDialog.show("Gmail Login");
50   * </pre></blockquote>
51   * 
52   * @see Credentials
53   * @author Tomas Varaneckas &lt;tomas.varaneckas@gmail.com&gt;
54   * @version $Id: LoginDialog.java 37 2009-07-09 17:26:41Z tomas.varaneckas $
55   * @since 0.1
56   */
57  public class LoginDialog {
58         
59      /**
60       * Dialog frame
61       */
62      private JFrame dialog;
63      
64      /**
65       * User text input
66       */
67      private JTextField user;
68      
69      /**
70       * Password input
71       */
72      private JPasswordField pass;

74      /**
75       * Username label
76       */
77      private JLabel labelUser;
78      
79      /**
80       * Password label

82      private JLabel labelPass;

84      /**
85       * OK button
86       */
87      private JButton ok;
88  
89      /**
90       * Countdown latch to make this dialog application modal
91       */
92      private CountDownLatch latch;

97      private static LoginDialog instance = null;

107                 instance = new LoginDialog();

116     public static final Log log = LogFactory.getLog(LoginDialog.class);

118     /**
119      * Method that shows login window with prompt for username and password,
120      * like this:
121      * <p><pre>
122      * +------ prompt ------+
123      * |Username            |
124      * |+------------------+|
125      * ||                  ||
126      * |+------------------+|
127      * |Password            |
128      * |+------------------+|
129      * ||                  ||
130      * |+------------------+|
131      * |[OK]                |
132      * +--------------------+
133      * </pre>
134      * @param prompt
135      * @return Credentials with username/password from the dialog prompt
136      */
137     public Credentials show(final String prompt) {
138         createDialog(prompt);
139         final Credentials login = new Credentials();
140         createListeners(login);
141         showDialog();
142         waitForData();
143         return login;
144     }
145 
146     /**
147      * Creates the required listeners
148      * 
149      * @param login Credentials object for filling the data
150      */
151     private void createListeners(final Credentials login) {
152         pass.addKeyListener(new KeyAdapter() {
153             @Override
154             public void keyReleased(final KeyEvent e) {
155                 if (e.getKeyCode() == KeyEvent.VK_ENTER) {
156                     setCredentials(login);
157                     dialog.setVisible(false);
158                 }
159             } 
160         });
161         ok.addActionListener(new ActionListener() {
162             public void actionPerformed(final ActionEvent e) {
163                 setCredentials(login);
164                 dialog.setVisible(false);
165             }
166         });
167         dialog.addWindowListener(new WindowAdapter() {
168             @Override
169             public void windowClosing(final WindowEvent e) {
170                 log.warn("Login dialog closed, no username/password specified");
171                 latch.countDown();
172             }
173         });
174     }
175 
176     /**
177      * Lets latch to wait for data
178      */
179     private void waitForData() {
180         try {
181             log.info("Waiting for username/password");
182             latch.await(30, TimeUnit.SECONDS);
183         } catch (final InterruptedException e1) {
184             log.warn("Interrupted", e1);
185         }
186     }
187 
188     /**
189      * Creates the dialog
190      * 
191      * @param prompt Dialog prompt
192      */
193     private void createDialog(final String prompt) {
194         dialog = new JFrame(prompt);
195         dialog.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
196         latch = new CountDownLatch(1);
197         labelUser = new JLabel("Username");
198         labelPass = new JLabel("Password");
199         user = new JTextField();
200         pass = new JPasswordField();
201         ok = new JButton("OK");
202     }
203 
204     /**
205      * Shows the dialog
206      */
207     private void showDialog() {
208         final BoxLayout layout = new BoxLayout(dialog.getContentPane(), 
209                 BoxLayout.Y_AXIS);
210         dialog.getContentPane().setLayout(layout);
211         dialog.getContentPane().add(labelUser);
212         dialog.getContentPane().add(user);
213         dialog.getContentPane().add(labelPass);
214         dialog.getContentPane().add(pass);
215         dialog.getContentPane().add(ok);
216         dialog.pack();
217         dialog.setSize(200, dialog.getSize().height);
218         dialog.setResizable(false);
219         dialog.setAlwaysOnTop(true);
220         center(dialog);
221         dialog.setVisible(true);
222         user.requestFocusInWindow();
223     }
224 
225     /**
226      * Centers the frame on screen
227      * 
228      * @param frame Target frame
229      */
230     private void center(final JFrame frame) {
231         final Rectangle screen = GraphicsEnvironment.getLocalGraphicsEnvironment()
232                 .getDefaultScreenDevice().getDefaultConfiguration().getBounds();
233         frame.setLocation(screen.width / 2 - dialog.getSize().width / 2, 
234                 screen.height / 2 - dialog.getSize().height / 2);
235     }
236 
237     /**
238      * Inner method that releases the {@link CountDownLatch} and fills 
239      * {@link Credentials} object with values from Username and Password fields
240      * 
241      * @param user User {@link JTextField}
242      * @param pass Password {@link JPasswordField}
243      * @param login Login {@link Credentials}
244      */
245     private void setCredentials(final Credentials login) {
246         login.setUsername(user.getText());
247         login.setPassword(pass.getPassword());
248         latch.countDown();
249     }
